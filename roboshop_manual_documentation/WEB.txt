1. Checking if Nginx is available or not
	apt list --installed 2>/dev/null | grep nginx 1>/dev/null
2. If not available install Nginx	
	install nginx -y
3. rename existing Nginx file (for backup)
	mv /etc/nginx/nginx.conf /etc/nginx/nginx.backup.conf
4. check /frontend directory available or not, if not create
5. check if static data available or not, if not move static data to frontend directory
	git clone https://github.com/instana/robot-shop.git /tmp/robot-shop
	cp -r /tmp/robot-shop/web/static/ /frontend
6. create new Nginx file with the following info
	echo 'http {
    include /etc/nginx/mime.types;

    server {
        listen 80;
        server_name localhost;

        root /frontend/static/;

        add_header Cache-Control public;
        add_header Pragma public;
        expires 1M;
    }
}' >> /etc/nginx/nginx.conf
7. daemon-reload
	systemctl daemon-reload
8. enable nginx service
	systemctl enable nginx
9. start nginx service
	systemctl start nginx
